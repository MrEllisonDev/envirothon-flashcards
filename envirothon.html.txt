function renderHome(){
  const tGot=Object.values(allScores).reduce((a,s)=>a+(s.got||0),0);
  const tMissed=Object.values(allScores).reduce((a,s)=>a+(s.missed||0),0);
  const tSeen=Object.keys(allScores).length;
  const pd=document.getElementById("home-progress");
  if(tGot+tMissed>0){
    pd.style.display="flex";
    document.getElementById("home-stats").innerHTML=
      `<span style="color:#1a5c1a;font-weight:700">✓ ${tGot} correct</span>&nbsp;·&nbsp;<span style="color:#8b1a1a;font-weight:700">✗ ${tMissed} missed</span>&nbsp;·&nbsp;${tSeen} cards seen`;
  }else{pd.style.display="none";}

  const imgCount=ALL_CARDS.filter(c=>getImg(c)).length;
  const badge=document.getElementById("img-count-badge");
  badge.textContent=imgCount+" / "+ALL_CARDS.length;
  badge.style.background=imgCount>0?"#2d6a2d":"#bbb";

  const dl=document.getElementById("deck-list");dl.innerHTML="";
  Object.entries(DECK_GROUPS).forEach(([gk,g])=>{
    const block=document.createElement("div");block.className="deck-block";
    let catsHtml="";
    Object.entries(g.cats).forEach(([ck,label])=>{
      const cards=getCardsForCat(gk,ck);
      const mastered=cards.filter(c=>{const s=allScores[c.id];return s&&s.got>0&&s.got>=s.missed;}).length;
      const attempted=cards.filter(c=>allScores[c.id]).length;
      const sub=attempted>0?`<div class="mastered">${mastered}/${cards.length} mastered</div>`:"";
      catsHtml+=`<button class="cat-btn" style="background:${g.color}18;color:${g.color};border-color:${g.color}40" onclick="startDeck('${gk}','${ck}')">${label}${sub}</button>`;
    });
    block.innerHTML=`<div class="deck-header" style="background:${g.color}"><div class="deck-title">${g.label}</div><div class="deck-desc">${g.desc}</div></div><div class="deck-cats">${catsHtml}</div>`;
    dl.appendChild(block);
  });
}

}

function getCardsForCat(gk,ck){
  const src=gk==="nature"?NATURE_CARDS:NPS_CARDS;
  return(ck==="all_nature"||ck==="all_nps")?[...src]:src.filter(c=>c.cat===ck);
}

  activeDeck=gk;activeCat=ck;

}

  aa.style.borderTopColor=color+"40";
  aa.style.background=color+"08";
  aa.innerHTML=`<div class="answer-name" style="color:${color}">✓ ${card.name}</div><div class="answer-body">${card.back}</div>`;
  document.getElementById("card-controls").innerHTML=`<button class="btn btn-reveal" style="background:${color}" onclick="revealCard()">Reveal Answer</button>`;
  flipped=false;
}

function revealCard(){
  flipped=true;
  document.getElementById("card-answer").style.display="block";
  document.getElementById("card-controls").innerHTML=`<div class="btn-row"><button class="btn-wrong" onclick="nextCard(false)">✗ Still Learning</button><button class="btn-right" onclick="nextCard(true)">✓ Got It</button></div>`;
}

function nextCard(got){
  const card=deck[deckIdx];
  sessionScores[card.id]=got;
  allScores[card.id]={
    got:(allScores[card.id]?.got||0)+(got?1:0),
    missed:(allScores[card.id]?.missed||0)+(got?0:1)
  };
  saveScores();advanceCard();
}
function skipCard(){advanceCard();}
function advanceCard(){
  if(deckIdx+1<deck.length){deckIdx++;renderCard();}
  else showDone();
}

function showDone(){
  const got=Object.values(sessionScores).filter(Boolean).length;
  const total=Object.keys(sessionScores).length;
  const tGot=Object.values(allScores).reduce((a,s)=>a+(s.got||0),0);
  const tMissed=Object.values(allScores).reduce((a,s)=>a+(s.missed||0),0);
  document.getElementById("done-session").textContent=got+" / "+total+" correct this round";
  document.getElementById("done-alltime").innerHTML=
    `<span style="color:#1a5c1a;font-weight:700">✓ ${tGot} correct</span>&nbsp;·&nbsp;<span style="color:#8b1a1a;font-weight:700">✗ ${tMissed} missed</span>`;
  showScreen("done");
}

function renderManage(){
  const list=document.getElementById("manage-list");list.innerHTML="";
  ALL_CARDS.forEach(card=>{
    const col=CAT_COLORS[card.cat]||"#555";
    const imgSrc=getImg(card);
    const thumbHtml=imgSrc?`<img src="${imgSrc}" alt="${card.name}">`:`<span>${CAT_EMOJI[card.cat]}</span>`;
    const removeBtn=imgSrc?`<button class="btn-remove" onclick="removeImage(${card.id})">✕</button>`:"";
    const row=document.createElement("div");row.className="manage-row";row.id="manage-row-"+card.id;
    row.innerHTML=`<div class="manage-thumb">${thumbHtml}</div><div style="flex:1;min-width:0"><div class="manage-name">${card.name}</div><div class="manage-cat" style="color:${col}">${CAT_EMOJI[card.cat]} ${card.cat.replace("_"," ")}</div></div><div class="manage-btns"><button class="btn-add" style="background:${col}" onclick="triggerUpload(${card.id})">${imgSrc?"Replace":"Add"}</button>${removeBtn}</div>`;
    list.appendChild(row);
  });
}

function triggerUpload(cardId){
  pendingUploadId=cardId;
  const fi=document.getElementById("file-input");
  fi.value="";fi.click();
}
function removeImage(cardId){
  delete userImages[cardId];saveUserImages();
  if(currentScreen==="manage")renderManage();
  if(currentScreen==="card")renderCard();
  renderHome();
}

document.getElementById("file-input").addEventListener("change",function(e){
  const file=e.target.files[0];
  if(!file||!pendingUploadId)return;
  const reader=new FileReader();
  reader.onload=function(ev){
    userImages[pendingUploadId]=ev.target.result;
    saveUserImages();
    if(currentScreen==="manage")renderManage();
    if(currentScreen==="card")renderCard();
    renderHome();
  };
  reader.readAsDataURL(file);
  e.target.value="";
});

renderHome();
